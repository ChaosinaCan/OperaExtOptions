  4 /**
  5  * @namespace
  6  * Object that automates loading and saving settings in an Opera extension's options page
  7  * 
  8  * @property {SettingStorage} storage the SettingStorage object used to initialize the page
  9  */
 10 var OptionsPage = new function OptionsPage() {
 11 	
 12 	var storage = null;
 13 	var initialized = false;
 14 	
 15 	this.__defineGetter__('storage', function() {return storage});
 16 	
 17 	// list of FORM elements
 18 	var formElements;
 19 	
 20 	// types of form elements
 21 	// skip: ignore these. 
 22 	// checkable: element has the "checked" property
 23 	// radio: setting value should be calculated via getRadioValue()
 24 	var skip      = hash('hidden,submit,image,file,reset,button');
 25 	var numeric   = hash('number,range');
 26 	var checkable = hash('checkbox,radio');
 27 	var radio     = hash('radio');
 28 	
 29 	
 30 	
 31 	/**
 32 	 * @private
 33 	 * Turns a string into a hash table
 34 	 */
 35 	function hash(str, sep) {
 36 		var obj = {};
 37 		var tmp = str.split(sep || ',');
 38 
 39 		while( tmp.length ) 
 40 			obj[ tmp.pop() ] = true;
 41 
 42 		return obj;
 43 	}
 44 	
 45 	/**
 46 	 * @private
 47 	 * Returns true if the element is a <input> or <select> element
 48 	 */
 49 	function isInput(element) {
 50 		var tagname = element.tagName.toLowerCase();
 51 		return tagname == 'input' || tagname == 'select';
 52 	}
 53 	
 54 	/**
 55 	 * @private
 56 	 * Gets the value of a set of radio buttons
 57 	 */
 58 	function getRadioValue(element) {
 59 		var name = element.name;
 60 		var inputs = element.form ? element.form.elements[name] : 
 61 			document.querySelectorAll('input[type=radio][name="' + name + '"]');
 62 		for (var i = 0; i < inputs.length; i++) {
 63 			if (inputs[i].checked)
 64 				return inputs[i].value;
 65 		}
 66 		return null;
 67 	}
 68 	
 69 	
 70 	/**
 71 	 * @private
 72 	 * walk the elements and apply a callback method to them
 73 	 */
 74 	function walkElements(callback)
 75 	{
 76 		var obj = [];
 77 		for(var i = 0, element = null; element = formElements[i]; i++)
 78 		{
 79 			// skip the element if it has no name, is of a type with no useful value,
 80 			// or has the data-nosave attribute set
 81 			var type = element.type.toLowerCase();
 82 			var name = element.name || '';
 83 			if(skip[type] === true || name == '' || element.hasAttribute('data-nosave')) 
 84 				continue;
 85 
 86 			var tmp = callback(element, name, type);
 87 			if(tmp != null)
 88 				obj.push(tmp);
 89 		}
 90 		return obj;
 91 	}
 92 	
 93 	/**
 94 	 * @private 
 95 	 * sets the value of an element to its saved value
 96 	 */
 97 	function updateElement(element, name, type) {
 98 		var name = name || element.name || '';
 99 		var type = type || element.type.toLowerCase();
100 		var value = storage.get(name);
101 		if(checkable[type] === true) {
102 			if (radio[type] === true)
103 				element.checked = (element.value == value);
104 			else
105 				element.checked = value;
106 		}
107 		else {
108 			element.value = value;
109 			if (numeric[type] === true && coerceToLimits(element) != value)
110 				elementChanged(element);
111 		}
112 	}
113 	
114 	/**
115 	 * @private 
116 	 * Fixes the value of a control if it is outside of its min/max
117 	 */
118 	function coerceToLimits(element, value) {
119 		var value = value || element.valueAsNumber;
120 		if (element.min && value < parseFloat(element.min)) 
121 			value = element.value = element.min;
122 		if (element.max && value > parseFloat(element.max)) 
123 			value = element.value = element.max;
124 		
125 		// coerce NaN to 0
126 		if (isNaN(value))
127 			value = element.value = 0;
128 		
129 		return value;
130 	}
131 	
132 	/**
133 	 * @private
134 	 * save the new value of an element
135 	 */
136 	function elementChanged(e) {
137 		var element = e.currentTarget || e;
138 		var type = element.type.toLowerCase();
139 		var name = element.name || '';
140 		var value;
141 		if (checkable[type] === true)
142 			value = (radio[type] === true) ? getRadioValue(element) : element.checked;
143 		else if (numeric[type] === true) {
144 			value = element.valueAsNumber;
145 			value = coerceToLimits(element, value);
146 		}
147 		else
148 			value = element.value;
149 		
150 		storage.set(name, value);	
151 	}
152 	
153 	/**
154 	 * @private
155 	 */
156 	function setupElement(element, name, type) {
157 		var type = type || element.type.toLowerCase();
158 		var name = name || element.name || '';
159 		
160 		element.addEventListener('change', elementChanged, true);
161 		updateElement(element, name, type);
162 	}
163 	
164 	/**
165 	 * @private 
166 	 * Makes a button reset the value of an element to its default
167 	 */
168 	function addResetButton(button, element) {
169 		button.addEventListener('click', function() {
170 			storage.reset(element.name);
171 			updateElement(element);
172 		}, false);
173 	}
174 	
175 	/**
176 	 * @private 
177 	 * Finds all reset buttons and attaches them to their settings
178 	 */
179 	function setupAllResetButtons() {
180 		var resets = document.querySelectorAll('[data-resets]');
181 		for (var i = 0; i < resets.length; i++) {
182 			var elements = document.getElementsByName(resets[i].dataset.resets);
183 			for (var j = 0; j < elements.length; j++) 
184 				addResetButton(resets[i], elements[j]);
185 		}
186 	}
187 	
188 	/**
189 	 * @private
190 	 * Finds a reset button linked to a setting
191 	 */
192 	function findResetButton(name) {
193 		try {
194 			return document.querySelector('[data-resets="' + name + '"]');
195 		}
196 		catch (e) {
197 			return null;
198 		}
199 	}
200 	
201 	/**
202 	 * @private
203 	 */
204 	function onDOMContentLoaded() {		
205 		// walk and set the elements accordingly to the storage
206 		formElements = document.querySelectorAll('input,select')
207 		walkElements(setupElement);
208 		
209 		setupAllResetButtons();
210 		
211 		initialized = true;
212 	}

217 	/**
218 	 * Initializes the options page
219 	 * @param {SettingStorage} [settingstorage] The storage object to use when 
220 	 *		loading and saving settings. If unused, a new SettingStorage object
221 	 *		with default settings will be used.
222 	 */
223 	this.init = function(settingstorage) {
224 		storage = settingstorage || new SettingStorage();
225 		
226 		window.addEventListener('DOMContentLoaded', onDOMContentLoaded, false);
227 	}
228 	
229 	/**
230 	 * Updates the value of an element to its value saved in storage
231 	 * @param {String|HTMLElement|HTMLElement[]} element The name of a setting, a
232 	 *		specific element to update, or a list of elements to update
233 	 */
234 	this.update = function(element) {
235 		if (!element)
236 			return;
237 		
238 		if (typeof element === 'string')
239 			element = document.getElementsByName(element);
240 		if (element.length === undefined)
241 			updateElement(element);
242 		else
243 			for (var i = 0; i < element.length; i++)
244 				updateElement(element[i]);
245 	}
246 	
247 	/**
248 	 * Saves the current value of an element to storage
249 	 * @param {String|HTMLElement|HTMLElement[]} element The name of a setting, a
250 	 *		specific element to save, or a list of elements to save
251 	 */
252 	this.save = function(element) {
253 		if (!element)
254 			return;
255 		
256 		if (typeof element === 'string')
257 			element = document.querySelector('[name="' + element + '"]');
258 		if (element.length === undefined)
259 			elementChanged(element);
260 		else
261 			for (var i = 0; i < element.length; i++)
262 				elementChanged(element[i]);
263 	}
264 	
265 	/**
266 	 * Sets up a new <input> or <select> element. Use this if an element is
267 	 * dynamically added after the options page loads.
268 	 * @param {HTMLElement} element The <input> or <select> element to add
269 	 * @param {HTMLElement} [resetbutton] A <button> or other clickable element
270 	 *		that should reset the setting when clicked. If this parameter is unused and a
271 	 *		reset button for the setting already exists on the page, it will be used.
272 	 */
273 	this.addInput = function(element, resetbutton) {
274 		if (!initialized) 
275 			throw new Error('OptionsPage.addInput cannot be called before the DOMContentLoaded event has fired.')
276 		
277 		if (element.tagName && isInput(element)) {
278 			setupElement(element);
279 			
280 			var reset = resetbutton || findResetButton(element.name);
281 			if (reset)
282 				addResetButton(reset, element);
283 		}
284 	}
285 	
286 	/**
287 	 * Returns a <dl> element containing each of the key-value pairs stored in 
288 	 * the web storage object used by the options page's SettingStorage object.
289 	 * @param {Function} [sortfunction] A function used to sort the list. If 
290 	 *		unused, the list will be sorted alphabetically.
291 	 */
292 	this.debugStorage = function(sortfunction) {
293 		var keys = [];
294 		for (var key in storage.storage)
295 			keys.push(key);
296 		
297 		keys.sort(sortfunction);
298 		
299 		var list = document.createElement('dl');
300 		for (var i = 0; i < keys.length; i++) {
301 			var term = document.createElement('dt');
302 			var desc = document.createElement('dd');
303 			term.textContent = keys[i];
304 			desc.textContent = storage.storage[keys[i]];
305 			list.appendChild(term);
306 			list.appendChild(desc);
307 		}
308 		
309 		return list;
310 	}
