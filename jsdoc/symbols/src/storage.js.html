  4 if ( !Function.prototype.bind ) {
  5 
  6 	Function.prototype.bind = function( obj ) {
  7 		if(typeof this !== 'function') // closest thing possible to the ECMAScript 5 internal IsCallable function
  8 			throw new TypeError('Function.prototype.bind - what is trying to be bound is not callable');
  9 
 10 		var slice = [].slice,
 11 			args = slice.call(arguments, 1), 
 12 			self = this, 
 13 			nop = function () {}, 
 14 			bound = function () {
 15 				return self.apply( this instanceof nop ? this : ( obj || {} ), 
 16 					args.concat( slice.call(arguments) ) );    
 17 			};
 18 
 19 		bound.prototype = this.prototype;
 20 
 21 		return bound;
 22 	};

 26 /**
 27  * @class Abstracts a web storage object so that arbitrary objects can be saved via JSON.
 28  * @property {bool} firstRun true if this is the first time the extension has been
 29  *		run (this.init() has never been called for this object)
 30  * 
 31  * @constructor
 32  * @param [defaults] An array of default setting values. Each default value 
 33  *		should be a two-element array where index 0 contains the setting name and 
 34  *		index 1 contains its default value. ex: [['setting 1', null], ['setting 2', 'foo']]
 35  *	@param [options] An object containing extra options
 36  *	@param [options.initSetting] The name of a setting used to determine whether 
 37  *		storage has been initialized. If storage has not been initialized, all
 38  *		settings with default values will be set to their defaults and initSetting
 39  *		will be set to "true". Defaults to "__initialized". 
 40  *	@param [options.prefix] A string to place before all setting names. This can 
 41  *		be used to allow multiple SettingStorage objects in one web storage space.
 42  *	@param [options.globalInit] Set to "true" to make initSetting ignore the setting
 43  *		prefix. This option mostly exists to make this library compatible with 
 44  *		Tab Vault's storage format.
 45  *	@param [options.useAccessors] Set to "false" to disable creating accessor
 46  *		properties for each setting. You should disable this if you plan to support
 47  *		old versions of Opera which do not support Object.defineProperties().
 48  *	@param [options.storage] The web storage object to use. Defaults to widget.preferences
 49  */
 50 function SettingStorage() {
 51 	// defaults: [name, defaultValue]
 52 	this.defaults = [];
 53 	this.prefix = '';
 54 	this.storage = widget.preferences;
 55 	this.initSetting = '__initialized';
 56 	this.globalInit = false;
 57 	this.useAccessors = true;
 58 
 59 	var firstRun = false;
 60 	this.__defineGetter__('firstRun', function() {return firstRun});
 61 
 62 	/**
 63 	 * @private
 64 	 */
 65 	function isDefined(value) {
 66 		return value !== undefined;
 67 	}
 68 	
 69 	/**
 70 	 * @private
 71 	 * Initializes settings
 72 	 */
 73 	function init(defaults, options) {
 74 		if (isDefined(defaults))
 75 			this.defaults = defaults || [];
 76 		
 77 		if (isDefined(options)) {
 78 			if (isDefined(options.prefix))
 79 				this.prefix = options.prefix;
 80 			if (isDefined(options.storage))
 81 				this.storage = options.storage;
 82 			if (isDefined(options.initSetting))
 83 				this.initSetting = options.initSetting;
 84 			if (isDefined(options.globalInit))
 85 				this.globalInit = options.globalInit;
 86 			if (isDefined(options.useAccessors))
 87 				this.useAccessors = options.useAccessors;
 88 		}
 89 		
 90 		if (!getInit.bind(this)())
 91 			firstRun = true;
 92 		
 93 		if (this.useAccessors)
 94 			defineAccessors.bind(this)();
 95 	}
 96 	
 97 	/**
 98 	 * @private
 99 	 * Gets the init flag
100 	 */
101 	function getInit() {
102 		if (this.globalInit) {
103 			var temp = this.prefix;
104 			this.prefix = '';
105 			var init = this.get(this.initSetting);
106 			this.prefix = temp;
107 			return init;
108 		}
109 		else return this.get(this.initSetting);
110 	}
111 	
112 	/**
113 	 * @private
114 	 * Sets the init flag
115 	 */
116 	function setInit(value) {
117 		if (this.globalInit) {
118 			var temp = this.prefix;
119 			this.prefix = '';
120 			this.set(this.initSetting, value);
121 			this.prefix = temp;
122 		}
123 		else
124 			this.set(this.initSetting, value);
125 	}
126 
127 	/**
128 	 *@private
129 	 * Defines properties so settings can be referred to by name
130 	 */
131 	function defineAccessors() {
132 		var descriptors = {};
133 		var reserved = ['init', 'reset', 'resetAll', 'fillDefaults', 'get', 'set', 'getAll', 'setAll'];
134 		
135 		function desc(name) {
136 			return {
137 				get: function() { return this.get(name) },
138 				set: function(x) { this.set(name, x) },
139 				enumerable: true,
140 			}
141 		}
142 		
143 		for (var i = 0; i < this.defaults.length; i++) {
144 			var name = this.defaults[i][0];
145 			if (reserved.indexOf(name) == -1)
146 				descriptors[name] = desc(name);
147 		}
148 		
149 		Object.defineProperties(this, descriptors);
150 	}
151 
152 
153 	/**
154 	 * Initializes the storage object. This should be called inside the 
155 	 * background script only and before any other calls to this object are made.
156 	 */
157 	this.init = function() {
158 		// Fill all uninitialized settings with their default values
159 		// If 1st time, set the initialized flag.	
160 		this.fillDefaults();
161 		if (this.firstRun)
162 			setInit.bind(this)(true);
163 	}
164 
165 	/**
166 	 * Resets a setting to its default value
167 	 * @param {String} name The setting to reset
168 	 */
169 	this.reset = function(name) {
170 		for (var i = 0; i < this.defaults.length; i++) {
171 			if (this.defaults[i][0] == name) {
172 				this.set(name, this.defaults[i][1]);
173 				return;
174 			}
175 		}
176 		this.set(name, null);
177 	}
178 
179 	/**
180 	 * Resets all settings to their default values
181 	 */
182 	this.resetAll = function() {
183 		for (var i = 0; i < this.defaults.length; i++)
184 			this.set(this.defaults[i][0], this.defaults[i][1]);
185 	}
186 
187 	/**
188 	 * Finds any undefined settings and fills them with their default values
189 	 */
190 	this.fillDefaults = function() {
191 		for (var i = 0; i < this.defaults.length; i++) {
192 			if (!isDefined(this.storage[this.prefix + this.defaults[i][0]]))
193 				this.set(this.defaults[i][0], this.defaults[i][1]);
194 		}
195 	}
196 
197 	/**
198 	 * Gets the value of a setting
199 	 * @param {String} name The setting to get
200 	 */
201 	this.get = function(name) {
202 		var data = this.storage[this.prefix + name];
203 		if (!isDefined(data))
204 			return null;
205 		return JSON.parse(data);
206 	}
207 
208 	/**
209 	 * Sets the value of a setting
210 	 * @param {String} name The setting to set
211 	 * @param value The value to save
212 	 */
213 	this.set = function(name, value) {
214 		var temp = JSON.stringify(value);
215 		this.storage[this.prefix + name] = temp;
216 	}
217 
218 	/**
219 	 * Gets an object with the values of all settings as key-value pairs.
220 	 * This only retrieves settings that have defined default values.
221 	 */
222 	this.getAll = function() {
223 		var data = {};
224 		for (var i = 0; i < this.defaults.length; i++) 
225 			data[this.defaults[i][0]] = this.get(this.defaults[i][0]);
226 		return data;
227 	}
228 
229 	/**
230 	 * Sets all the settings defined in the data object
231 	 * @param {Object} data An object containing the names and values of settings
232 	 *		as key-value pairs
233 	 */
234 	this.setAll = function(data) {
235 		for (var name in data) 
236 			this.set(name, data[name]);
237 	}
238 	
239 	
240 	init.bind(this)(arguments[0], arguments[1]);
241 }
